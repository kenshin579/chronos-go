```go
// GroupMemberIDs returns the IDs of a group's not-yet-succeeded members
// (empty when the group finished or its record expired).
func (r *RDB) GroupMemberIDs(ctx context.Context, cbQueue, groupID string) ([]string, error) {
	return r.client.SMembers(ctx, base.GroupKey(cbQueue, groupID)).Result()
}

// LeaderID returns the current scheduler leader's instance ID ("" when no
// leader holds the lock).
func (r *RDB) LeaderID(ctx context.Context) (string, error) {
	id, err := r.client.Get(ctx, base.LeaderKey()).Result()
	if err == redis.Nil {
		return "", nil
	}
	return id, err
}

// ScheduleFireInfo is one schedule that has fired at least once.
type ScheduleFireInfo struct {
	ID        string
	LastFired int64 // unix seconds
}

// ScanSchedules lists schedules that have ever fired, with their last-fired
// times, by scanning the per-schedule last-fired keys. Registered-but-never-
// fired schedules are invisible (they live only in scheduler process memory).
func (r *RDB) ScanSchedules(ctx context.Context) ([]ScheduleFireInfo, error) {
	const prefix = "chronos:sched:"
	const suffix = ":last"
	var out []ScheduleFireInfo
	iter := r.client.Scan(ctx, 0, prefix+"*"+suffix, 100).Iterator()
	for iter.Next(ctx) {
		key := iter.Val()
		raw, err := r.client.Get(ctx, key).Result()
		if err == redis.Nil {
			continue
		}
		if err != nil {
			return nil, err
		}
		ts, err := strconv.ParseInt(raw, 10, 64)
		if err != nil {
			continue // foreign key shape; skip
		}
		id := strings.TrimSuffix(strings.TrimPrefix(key, prefix), suffix)
		out = append(out, ScheduleFireInfo{ID: id, LastFired: ts})
	}
	if err := iter.Err(); err != nil {
		return nil, err
	}
	return out, nil
}
```
(import에 `strconv` 추가; `strings`는 이미 있음. **Cluster 주의**: `Scan`은 go-redis ClusterClient에서 단일 노드만 스캔 — `chronos:sched:*` 키는 해시태그가 없어 여러 슬롯에 분산됨. UniversalClient가 *redis.ClusterClient면 `ForEachMaster`로 전 노드 스캔하도록 분기:
```go
	if cc, ok := r.client.(*redis.ClusterClient); ok {
		var mu sync.Mutex
		err := cc.ForEachMaster(ctx, func(ctx context.Context, m *redis.Client) error {
			iter := m.Scan(ctx, 0, prefix+"*"+suffix, 100).Iterator()
			for iter.Next(ctx) {
				... 동일 처리 (mu로 out 보호, GET은 r.client로 — 리다이렉트 처리)
			}
			return iter.Err()
		})
		...
	}
```
정확한 구조는 구현 시 판단하되 standalone/cluster 둘 다 동작해야 함 — cluster 스모크가 검증.)

```go
// GroupMembers returns the IDs of a group's not-yet-succeeded members. cbQueue
// is the callback queue (GroupInfo/TaskInfo carry it as the member's GroupQueue).
func (i *Inspector) GroupMembers(ctx context.Context, cbQueue, groupID string) ([]string, error) {
	return i.rdb.GroupMemberIDs(ctx, cbQueue, groupID)
}

// SchedulerStatus reports the current scheduler leader and every schedule that
// has fired at least once (registered-but-never-fired schedules are invisible
// until Phase 2's registry).
type SchedulerStatus struct {
	LeaderID  string
	Schedules []ScheduleInfo
}

// ScheduleInfo is one schedule's fire history.
type ScheduleInfo struct {
	ID        string
	LastFired time.Time
}

func (i *Inspector) SchedulerStatus(ctx context.Context) (*SchedulerStatus, error) {
	leader, err := i.rdb.LeaderID(ctx)
	if err != nil {
		return nil, err
	}
	raw, err := i.rdb.ScanSchedules(ctx)
	if err != nil {
		return nil, err
	}
	st := &SchedulerStatus{LeaderID: leader, Schedules: make([]ScheduleInfo, 0, len(raw))}
	for _, s := range raw {
		st.Schedules = append(st.Schedules, ScheduleInfo{ID: s.ID, LastFired: time.Unix(s.LastFired, 0)})
	}
	return st, nil
}
```
